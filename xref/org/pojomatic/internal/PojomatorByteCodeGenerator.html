98    private void acceptClassVisitor(ClassVisitor classWriter) {
99      classWriter.visit(V1_7, ACC_PUBLIC + ACC_SUPER + ACC_SYNTHETIC, pojomatorInternalClassName, null,
100         BASE_POJOMATOR_INTERNAL_NAME, new String[] { internalName(Pojomator.class) });
101 
102     classWriter.visitSource("Look for visitLineNumber", null);
103 
104     makeFields(classWriter);
105 
106     makeConstructor(classWriter);
107     makeBootstrapMethod(classWriter);
108 
109     for (PropertyElement propertyElement: classProperties.getAllProperties()) {
110       makeAccessor(classWriter, propertyElement);
111     }
112 
113     makeDoEquals(classWriter);
114     makeDoHashCode(classWriter);
115     makeDoToString(classWriter);
116     makeDoDiff(classWriter);
117 
118     classWriter.visitEnd();
119   }

137   /**
138    * Generate a bootstrap method that allows us to access properties without reflection, even though they may not be
139    * public. While the bulk of the work will be done by {@link BasePojomator}, we need to implement a shim which can
140    * provide our classname to
141    * {@link BasePojomator#bootstrap(java.lang.invoke.MethodHandles.Lookup, String, MethodType, Class)},
142    * so that it can in turn access the {@link PropertyElement} field on the class we are generating.
143    * @param classWriter
144    */
145   private void makeBootstrapMethod(ClassVisitor classWriter) {
146     LocalVariable caller = new LocalVariable("caller", MethodHandles.Lookup.class, null, 0);
147     LocalVariable name = new LocalVariable("name", String.class, null, 1);
148     LocalVariable type = new LocalVariable("type", MethodType.class, null, 2);
149     MethodVisitor mv = classWriter.visitMethod(
150        ACC_STATIC,
151        BOOTSTRAP_METHOD_NAME,
152        methodDesc(CallSite.class, MethodHandles.Lookup.class, String.class, MethodType.class),
153        null,
154        null);
155     mv.visitCode();
156     Label start = visitNewLabel(mv);
157 
158     caller.acceptLoad(mv);
159     name.acceptLoad(mv);
160     type.acceptLoad(mv);
161     visitLineNumber(mv, 1, null);
162     mv.visitLdcInsn(Type.getObjectType(pojomatorInternalClassName));
163     visitLineNumber(mv, 2, null);
164     mv.visitMethodInsn(
165       INVOKESTATIC, BASE_POJOMATOR_INTERNAL_NAME,
166       "bootstrap",
167       methodDesc(CallSite.class, MethodHandles.Lookup.class, String.class, MethodType.class, Class.class));
168     visitLineNumber(mv, 3, null);
169     mv.visitInsn(ARETURN);
170     Label end = visitNewLabel(mv);
171     caller.withScope(start, end).acceptLocalVariable(mv);
172     name.withScope(start, end).acceptLocalVariable(mv);
173     type.withScope(start, end).acceptLocalVariable(mv);
174     mv.visitMaxs(4, 3);
175     mv.visitEnd();
176   }

178   /**
179    * Generate an accessor method for a property. The generated method uses InvokeDynamic, calling the method generated
180    * by {@link #makeBootstrapMethod(ClassVisitor)}
181    * @param classWriter
182    * @param propertyElement the property to generate the accessor for
183    */
184   private void makeAccessor(ClassVisitor classWriter, PropertyElement propertyElement) {
185     LocalVariable pojo = new LocalVariable("pojo", Object.class, null, 0);
186     int maxStackSize = 1;
187     String accessorName = propertyAccessorName(propertyElement);
188     MethodVisitor mv = classWriter.visitMethod(
189       ACC_PRIVATE | ACC_STATIC, accessorName, accessorMethodDescription(propertyElement), null, null);
190     mv.visitCode();
191     Label start = visitNewLabel(mv);
192     pojo.acceptLoad(mv);
193     visitLineNumber(mv, 4, propertyElement);
194     mv.visitInvokeDynamicInsn(accessorName, accessorMethodDescription(propertyElement), bootstrapMethod);
195     visitLineNumber(mv, 5, propertyElement);
196 
197     // return using the appropriate return byte code, based on type
198     Class<?> propertyType = propertyElement.getPropertyType();
199     if (propertyType.isPrimitive()) {
200       if (propertyType == float.class) {
201         mv.visitInsn(FRETURN);
202       }
203       else if (propertyType == long.class) {
204         maxStackSize++;
205         mv.visitInsn(LRETURN);
206       }
207       else if (propertyType == double.class) {
208         maxStackSize++;
209         mv.visitInsn(DRETURN);
210       }
211       else {
212         mv.visitInsn(IRETURN);
213       }
214     }
215     else {
216       mv.visitInsn(ARETURN);
217     }
218 
219     Label end = visitNewLabel(mv);
220     pojo.withScope(start, end).acceptLocalVariable(mv);
221     mv.visitMaxs(maxStackSize, 2);
222     mv.visitEnd();
223   }

226     LocalVariable varThis = new LocalVariable("this", pojomatorInternalClassDesc, null, 0);
227     LocalVariable varPojoClass = new LocalVariable("pojoClass", Class.class, null, 1);
228     LocalVariable varClassProperties = new LocalVariable("classProperties", ClassProperties.class, null, 2);
229     MethodVisitor mv = cw.visitMethod(
230       ACC_PUBLIC, "<init>", methodDesc(void.class, Class.class, ClassProperties.class), null, null);

232     Label start = visitNewLabel(mv);
233     varThis.acceptLoad(mv);
234     varPojoClass.acceptLoad(mv);

237     mv.visitMethodInsn(
238       INVOKESPECIAL, BASE_POJOMATOR_INTERNAL_NAME, "<init>", methodDesc(void.class, Class.class, ClassProperties.class));
239     mv.visitInsn(RETURN);
240     Label end = visitNewLabel(mv);

244     mv.visitMaxs(3, 3);
245     mv.visitEnd();

248   /**
249    * Generate the {@link Pojomator#doEquals(Object, Object)} method.
250    * @param cw
251    */
252   private void makeDoEquals(ClassVisitor cw) {
253     LocalVariable varThis = new LocalVariable("this", pojomatorInternalClassDesc, null, 0);
254     LocalVariable varPojo1 = new LocalVariable("pojo1", pojoClass, pojoDescriptor, 1);
255     LocalVariable varPojo2 = new LocalVariable("pojo2", pojoClass, pojoDescriptor, 2);
256 
257     StackAdjustments stackAdjustments = new StackAdjustments();
258 
259     Object[] localVars = new Object[] {pojomatorInternalClassName, OBJECT_INTERNAL_NAME, OBJECT_INTERNAL_NAME};
260 
261     MethodVisitor mv = cw.visitMethod(
262       ACC_PUBLIC, "doEquals", methodDesc(boolean.class, Object.class, Object.class), null, null);
263 
264     // where to jump if we should return false
265     Label returnFalse = new Label();
266     // where to jump if we determine that pojo1 and pojo2 have types which are compatible for equality
267     Label compatibleTypes = new Label();
268 
269     mv.visitCode();
270     Label start = visitNewLabel(mv);
271     varPojo1.acceptLoad(mv);
272     visitLineNumber(mv, 7, null);
273     checkNotNull(mv);
274     varPojo2.acceptLoad(mv);
275     visitLineNumber(mv, 8, null);
276     Label notSameInstance = new Label();
277     mv.visitJumpInsn(IF_ACMPNE, notSameInstance);
278 
279     // same instance; return true
280     mv.visitInsn(ICONST_1);
281     mv.visitInsn(IRETURN);
282 
283     mv.visitLabel(notSameInstance);
284 
285     // if other is null, return false.
286     mv.visitFrame(F_FULL, 3, localVars, 0, NO_STACK);
287     visitLineNumber(mv, 9, null);
288     varPojo2.acceptLoad(mv);
289     mv.visitJumpInsn(IFNULL, returnFalse);
290 
291     // common case: if both types are the same, they are compatible for equality
292     varThis.acceptLoad(mv);
293     visitLineNumber(mv, 10, null);
294     invokeGetClass(mv);
295     varPojo1.acceptLoad(mv);
296     visitLineNumber(mv, 11, null);
297     invokeGetClass(mv);
298     mv.visitJumpInsn(IF_ACMPEQ, compatibleTypes);
299 
300     // types are not the same; check for compatibility
301     varThis.acceptLoad(mv);
302     varPojo2.acceptLoad(mv);
303     visitLineNumber(mv, 12, null);
304     invokeGetClass(mv);
305     visitLineNumber(mv, 13, null);
306     mv.visitMethodInsn(
307       INVOKEVIRTUAL, BASE_POJOMATOR_INTERNAL_NAME, "isCompatibleForEquality", methodDesc(boolean.class, Class.class));
308     mv.visitJumpInsn(IFEQ, returnFalse);
309 
310     // types are compatible, so start comparing properties
311     mv.visitLabel(compatibleTypes);
312     mv.visitFrame(F_FULL, 3, localVars, 0, NO_STACK);
313 
314     // Compare properties
315     for(PropertyElement propertyElement: classProperties.getHashCodeProperties()) {
316       visitLineNumber(mv, 14, propertyElement);
317       visitAccessorAndConvert(mv, varPojo1, propertyElement);
318       visitLineNumber(mv, 15, propertyElement);
319       visitAccessorAndConvert(mv, varPojo2, propertyElement);
320       visitLineNumber(mv, 16, propertyElement);
321       compareProperties(mv, returnFalse, propertyElement, stackAdjustments);
322     }
323     // If we have gotten this far, all properties are equal, so return true.
324     mv.visitInsn(ICONST_1);
325     mv.visitInsn(IRETURN);
326 
327     mv.visitLabel(returnFalse);
328     mv.visitFrame(F_FULL, 3, localVars, 0, NO_STACK);
329     mv.visitInsn(ICONST_0);
330     mv.visitInsn(IRETURN);
331 
332     Label end = visitNewLabel(mv);
333     varThis.withScope(start, end).acceptLocalVariable(mv);
334     varPojo1.withScope(start, end).acceptLocalVariable(mv);
335     varPojo2.withScope(start, end).acceptLocalVariable(mv);
336     mv.visitMaxs(2 + stackAdjustments.adjustments(2,  1), 3);
337     mv.visitEnd();
338   }

340   /**
341    * Compare a property from each pojo. It is assumed when this method is called that both property values have been
342    * loaded onto the stack. In the event the property value is a float or double, it is further assumed that it has
343    * been converted to an int or long.
344    * @param mv
345    * @param notEqualLabel where to jump if the property values are not equal
346    * @param propertyElement the property being compared
347    * @param stackAdjustments adjustments to be made to the max stack size, based on property type
348    * @return {@code true} if the value of the property will take two positions on the stack (i.e. is a long or double)
349    */
350   private void compareProperties(
351       MethodVisitor mv, Label notEqualLabel, PropertyElement propertyElement, StackAdjustments stackAdjustments) {
352     Class<?> propertyType = propertyElement.getPropertyType();
353     if (propertyType.isPrimitive()) {
354       if (isWide(propertyElement)) {
355         stackAdjustments.wideProperty = true;
356         mv.visitInsn(LCMP);
357         mv.visitJumpInsn(IFNE, notEqualLabel);
358       }
359       else {
360         mv.visitJumpInsn(IF_ICMPNE, notEqualLabel);
361       }
362     }
363     else {
364       if(propertyType.isArray()) {
365         Class<?> componentType = propertyType.getComponentType();
366         if (componentType.isPrimitive()) {
367           visitLineNumber(mv, 17, propertyElement);
368           mv.visitMethodInsn(
369             INVOKESTATIC,
370             internalName(Arrays.class),
371             "equals",
372             methodDesc(boolean.class, propertyType, propertyType));
373         }
374         else {
375           visitLineNumber(mv, 18, propertyElement);
376           mv.visitMethodInsn(
377             INVOKESTATIC,
378             BASE_POJOMATOR_INTERNAL_NAME,
379             "compareArrays",
380             methodDesc(boolean.class, Object.class, Object.class));
381         }
382       }
383       else {
384         if (isObjectPossiblyHoldingArray(propertyElement)) {
385           visitLineNumber(mv, 19, propertyElement);
386           mv.visitMethodInsn(
387             INVOKESTATIC,
388             BASE_POJOMATOR_INTERNAL_NAME,
389             "areObjectValuesEqual",
390             methodDesc(boolean.class, Object.class, Object.class));
391         }
392         else {
393           visitLineNumber(mv, 20, propertyElement);
394           mv.visitMethodInsn(
395             INVOKESTATIC,
396             internalName(Objects.class),
397             "equals",
398             methodDesc(boolean.class, Object.class, Object.class));
399         }
400       }
401       mv.visitJumpInsn(IFEQ, notEqualLabel);
402     }
403   }

405   /**
406    * Generate the {@link Pojomator#doHashCode(Object)} method.
407    * @param cw
408    */
409   private void makeDoHashCode(ClassVisitor cw) {
410     LocalVariable varThis = new LocalVariable("this", pojomatorInternalClassDesc, null, 0);
411     LocalVariable varPojo = new LocalVariable("pojo", pojoClass, pojoDescriptor, 1);
412 
413     int longOrDoubleStackAdjustment = 0;
414     Object[] localVars = new Object[] {pojomatorInternalClassName, OBJECT_INTERNAL_NAME};
415 
416     MethodVisitor mv = cw.visitMethod(ACC_PUBLIC, "doHashCode", methodDesc(int.class, Object.class), null, null);
417     mv.visitCode();
418     Label start = visitNewLabel(mv);
419     visitLineNumber(mv, 21, null);
420     varPojo.acceptLoad(mv);
421     visitLineNumber(mv, 22, null);
422     checkNotNullPop(mv);
423 
424     //algorithm:
425     // hashCode(prop_n) + 31 * (hashCode(prop_n-1) + 31 * ( ... (hashCode(prop_1) + 31 * 1) ... ))
426 
427     mv.visitInsn(ICONST_1); // this will just be multiplied by 31; let the optimizer take care of it
428 
429     for(PropertyElement propertyElement: classProperties.getHashCodeProperties()) {
430       // multiply what we have so far by 31.
431       visitLineNumber(mv, 23, propertyElement);
432       mv.visitIntInsn(BIPUSH, 31);
433       visitLineNumber(mv, 24, propertyElement);
434       mv.visitInsn(IMUL);
435 
436       visitLineNumber(mv, 25, propertyElement);
437       visitAccessorAndConvert(mv, varPojo, propertyElement); // grab the property value, converting a float or double
438       Class<?> propertyType = propertyElement.getPropertyType();
439       if (propertyType.isPrimitive()) {
440         // need to compute the hash code for this primitive value, based on its type
441         switch (propertyType.getName()) {
442           case "boolean":
443             visitLineNumber(mv, 26, propertyElement);
444             Label ifeq = new Label();
445             mv.visitJumpInsn(IFEQ, ifeq);
446             mv.visitIntInsn(SIPUSH, Boolean.TRUE.hashCode());
447             Label hashCodeDetermined = new Label();
448             mv.visitJumpInsn(GOTO, hashCodeDetermined);
449             mv.visitLabel(ifeq);
450             mv.visitFrame(F_FULL, 2, localVars, 1, new Object[] {INTEGER});
451             mv.visitIntInsn(SIPUSH, Boolean.FALSE.hashCode());
452             mv.visitLabel(hashCodeDetermined);
453             mv.visitFrame(F_FULL, 2, localVars, 2, new Object[] {INTEGER, INTEGER});
454             break;
455           case "byte":
456           case "char":
457           case "int":
458           case "short":
459           case "float":
460             break; // already an int (from the JVM's point of view)
461           case "double":
462           case "long":
463             longOrDoubleStackAdjustment = 3; // one extra for the field, two extra for the dup to do an xor
464 
465             // compute bits ^ (bits >> 32)
466 
467             visitLineNumber(mv, 27, propertyElement);
468             // we'll need a second copy to do the xor:
469             mv.visitInsn(DUP2);
470             // bitshift 32 right:
471             mv.visitIntInsn(BIPUSH, 32);
472             mv.visitInsn(LUSHR);
473             // xor with the original
474             mv.visitInsn(LXOR);
475             // chop of the high 32 bits
476             mv.visitInsn(L2I);
477             break;
478           default:
479             throw new IllegalStateException("unknown primitive type " + propertyType.getName());
480         }
481       }
482       else {
483         Label ifNonNull = new Label();
484         Label hashCodeDetermined = new Label();
485 
486         mv.visitInsn(DUP); // if it is non-null, let's not have to get it a second time.
487         mv.visitJumpInsn(IFNONNULL, ifNonNull);
488         // it's null
489         mv.visitInsn(POP); // won't need that duped copy after all
490         mv.visitInsn(ICONST_0);
491         mv.visitJumpInsn(GOTO, hashCodeDetermined);
492 
493         // it's not null
494         mv.visitLabel(ifNonNull);
495         mv.visitFrame(
496           F_FULL, 2, localVars, 2, new Object[] {INTEGER, Type.getInternalName(effectiveType(propertyType))});
497 
498         if(propertyType.isArray()) {
499           visitLineNumber(mv, 28, propertyElement);
500           mv.visitMethodInsn(
501             INVOKESTATIC,
502             internalName(Arrays.class),
503             isDeepArray(propertyElement) ? "deepHashCode" : "hashCode",
504             methodDesc(
505               int.class, propertyType.getComponentType().isPrimitive() ? propertyType : Object[].class)
506               );
507         }
508         else if (isObjectPossiblyHoldingArray(propertyElement)) {
509           // it *could* be an array; if so, we want to do an array hashCode.
510 
511           mv.visitInsn(DUP); // we'll still want the property value handy after calling getClass().isArray()
512           invokeGetClass(mv);
513           visitLineNumber(mv, 29, propertyElement);
514           mv.visitMethodInsn(INVOKEVIRTUAL, internalName(Class.class), "isArray", "()Z");
515           Label isArray = new Label();
516           mv.visitJumpInsn(IFNE, isArray); // if true
517 
518           // regular old hashCode
519           visitLineNumber(mv, 30, propertyElement);
520           mv.visitMethodInsn(INVOKEVIRTUAL, OBJECT_INTERNAL_NAME, "hashCode", "()I");
521           mv.visitJumpInsn(GOTO, hashCodeDetermined);
522 
523           // add a deep parameter to arrayHashCode, like we did for compareProperties
524           mv.visitLabel(isArray);
525           mv.visitFrame(F_FULL, 2, localVars, 2, new Object[] { INTEGER, Type.getInternalName(propertyType) });
526 
527           mv.visitInsn(ICONST_1);
528           visitLineNumber(mv, 31, propertyElement);
529           mv.visitMethodInsn(
530             INVOKESTATIC,
531             BASE_POJOMATOR_INTERNAL_NAME,
532             "arrayHashCode",
533             methodDesc(int.class, Object.class, boolean.class));
534         }
535         else {
536           visitLineNumber(mv, 32, propertyElement);
537           mv.visitMethodInsn(INVOKEVIRTUAL, OBJECT_INTERNAL_NAME, "hashCode", "()I");
538         }
539 
540         mv.visitLabel(hashCodeDetermined);
541         mv.visitFrame(F_FULL, 2, localVars, 2, new Object[] {INTEGER, INTEGER});
542       }
543       // add result to what we have so far
544       mv.visitInsn(IADD);
545     }
546     mv.visitInsn(IRETURN);
547     Label end = visitNewLabel(mv);
548     varThis.withScope(start, end).acceptLocalVariable(mv);
549     varPojo.withScope(start, end).acceptLocalVariable(mv);
550     mv.visitMaxs(3 + longOrDoubleStackAdjustment, 2);
551     mv.visitEnd();
552   }

554   /**
555    * Generate {@link Pojomator#doToString(Object)}
556    * @param cw
557    */
558   private void makeDoToString(ClassVisitor cw) {
559     int longOrDoubleStackAdjustment = 1;
560     LocalVariable varThis = new LocalVariable("this", pojomatorInternalClassDesc, null, 0);
561     LocalVariable varPojo = new LocalVariable("pojo", pojoClass, null, 1);
562     LocalVariable varPojoFormatter=
563       new LocalVariable("pojoFormattor", classDesc(EnhancedPojoFormatter.class), null, 2);
564     LocalVariable varBuilder= new LocalVariable("builder", classDesc(String.class), null, 3);
565 
566     MethodVisitor mv = cw.visitMethod(ACC_PUBLIC, "doToString", methodDesc(String.class, Object.class), null, null);
567     mv.visitCode();
568     Label start = visitNewLabel(mv);
569     varPojo.acceptLoad(mv);
570     checkNotNullPop(mv);
571 
572     constructEnhancedPojoFormatter(mv);
573     varPojoFormatter.acceptStore(mv);
574 
575     visitLineNumber(mv, 33, null);
576     mv.visitTypeInsn(NEW, internalName(StringBuilder.class));
577     mv.visitInsn(DUP);
578     mv.visitMethodInsn(INVOKESPECIAL, internalName(StringBuilder.class), "<init>", "()V");
579     varBuilder.acceptStore(mv);
580 
581     varPojoFormatter.acceptLoad(mv);
582     varBuilder.acceptLoad(mv);
583     loadPojoClass(mv);
584 
585     visitLineNumber(mv, 34, null);
586 
587     mv.visitMethodInsn(INVOKEINTERFACE, internalName(EnhancedPojoFormatter.class), "appendToStringPrefix",
588       methodDesc(void.class, StringBuilder.class, Class.class));
589 
590     for(PropertyElement propertyElement: classProperties.getToStringProperties()) {
591       if (isWide(propertyElement)) {
592         longOrDoubleStackAdjustment = 1; // having any double-wide values on our stack increases max stack depth by one
593       }
594 
595       // append the property prefix
596       varPojoFormatter.acceptLoad(mv);
597       varBuilder.acceptLoad(mv);
598       visitLineNumber(mv, 35, propertyElement);
599       loadPropertyElementField(mv, propertyElement);
600       visitLineNumber(mv, 36, propertyElement);
601       mv.visitMethodInsn(
602         INVOKEINTERFACE,
603         internalName(EnhancedPojoFormatter.class),
604         "appendPropertyPrefix",
605         methodDesc(void.class, StringBuilder.class, PropertyElement.class));
606 
607       // get the propertyFormatter for this property
608       visitLineNumber(mv, 37, propertyElement);
609       mv.visitFieldInsn(
610         GETSTATIC,
611         pojomatorInternalClassName,
612         propertyFormatterName(propertyElement),
613         classDesc(EnhancedPropertyFormatter.class));
614 
615       // The propertyFormatter will format the property value and append the results to our StringBuilder
616       varBuilder.acceptLoad(mv);
617       visitLineNumber(mv, 38, propertyElement);
618       visitAccessor(mv, varPojo, propertyElement);
619       if (isObjectPossiblyHoldingArray(propertyElement)) {
620         visitLineNumber(mv, 39, propertyElement);
621         mv.visitMethodInsn(
622           INVOKEINTERFACE,
623           internalName(EnhancedPropertyFormatter.class),
624           "appendFormattedPossibleArray",
625           methodDesc(
626             void.class,
627             StringBuilder.class,
628             appendFormattedType(propertyElement.getPropertyType()))
629           );
630       }
631       else {
632         visitLineNumber(mv, 40, propertyElement);
633         mv.visitMethodInsn(
634           INVOKEINTERFACE,
635           internalName(EnhancedPropertyFormatter.class),
636           "appendFormatted",
637           methodDesc(
638             void.class,
639             StringBuilder.class,
640             appendFormattedType(propertyElement.getPropertyType()))
641             );
642       }
643 
644       // have any property suffix appended to the StringBuilder
645       varPojoFormatter.acceptLoad(mv);
646       varBuilder.acceptLoad(mv);
647       visitLineNumber(mv, 41, propertyElement);
648       loadPropertyElementField(mv, propertyElement);
649       visitLineNumber(mv, 42, propertyElement);
650       mv.visitMethodInsn(
651         INVOKEINTERFACE,
652         internalName(EnhancedPojoFormatter.class),
653         "appendPropertySuffix",
654         methodDesc(void.class, StringBuilder.class, PropertyElement.class));
655     }
656 
657     // Have any toString suffix appended
658     varPojoFormatter.acceptLoad(mv);
659     varBuilder.acceptLoad(mv);
660     loadPojoClass(mv);
661     visitLineNumber(mv, 43, null);
662     mv.visitMethodInsn(INVOKEINTERFACE, internalName(EnhancedPojoFormatter.class), "appendToStringSuffix",
663       methodDesc(void.class, StringBuilder.class, Class.class));
664 
665     // invoke toString and return the result
666     varBuilder.acceptLoad(mv);
667     visitLineNumber(mv, 44, null);
668     mv.visitMethodInsn(INVOKEVIRTUAL, internalName(StringBuilder.class), "toString", "()Ljava/lang/String;");
669     mv.visitInsn(ARETURN);
670 
671     Label end = visitNewLabel(mv);
672     varThis.withScope(start, end).acceptLocalVariable(mv);
673     varPojo.withScope(start, end).acceptLocalVariable(mv);
674     varPojoFormatter.withScope(start, end).acceptLocalVariable(mv);
675     varBuilder.withScope(start, end).acceptLocalVariable(mv);
676     mv.visitMaxs(3 + longOrDoubleStackAdjustment, 4);
677     mv.visitEnd();
678   }

700   /**
701    * Construct the pojoFormatter to use. This method will contribute 2 or 4 to the max stack depth,
702    * depending on whether the pojoFormatter implements {@link EnhancedPojoFormatter} or not.
703    * @param mv
704    */
705   private void constructEnhancedPojoFormatter(MethodVisitor mv) {
706     PojoFormat format = pojoClass.getAnnotation(PojoFormat.class);
707     if (format == null) {
708       mv.visitTypeInsn(NEW, internalName(DefaultEnhancedPojoFormatter.class));
709       mv.visitInsn(DUP);
710       visitLineNumber(mv, 45, null);
711       mv.visitMethodInsn(INVOKESPECIAL, internalName(DefaultEnhancedPojoFormatter.class), "<init>", "()V");
712     }
713     else {
714       @SuppressWarnings("deprecation")
715       Class<? extends org.pojomatic.formatter.PojoFormatter> pojoFormatterClass = format.value();
716       // if it isn't an enhanced formatter, we'll need to wrap it a EnhancedPojoFormatter. If we do this, we'll first
717       // invoke new on the wrapper, then construct the underlying formatter, then call the constructor on the wrapper.
718       boolean isEnhancedFormatter = EnhancedPojoFormatter.class.isAssignableFrom(pojoFormatterClass);
719       if (! isEnhancedFormatter) {
720         visitLineNumber(mv, 46, null);
721         mv.visitTypeInsn(NEW, ENHANCED_POJO_FORMATTER_WRAPPER_INTERNAL_NAME);
722         mv.visitInsn(DUP);
723       }
724       mv.visitTypeInsn(NEW, internalName(pojoFormatterClass));
725       mv.visitInsn(DUP);
726       visitLineNumber(mv, 47, null);
727       mv.visitMethodInsn(INVOKESPECIAL, internalName(pojoFormatterClass), "<init>", "()V");
728       if (! isEnhancedFormatter) {
729         visitLineNumber(mv, 48, null);
730         mv.visitMethodInsn(
731           INVOKESPECIAL,
732           ENHANCED_POJO_FORMATTER_WRAPPER_INTERNAL_NAME,
733           "<init>",
734           ENHANCED_POJO_FORMATTER_WRAPPER_CONSTRUCTOR_DESCRIPTOR);
735       }
736     }

749   /**
750    * Generate {@link Pojomator#doDiff(Object, Object)
751    * @param cw
752    */
753   private void makeDoDiff(ClassVisitor cw) {
754     LocalVariable varThis = new LocalVariable("this", pojomatorInternalClassDesc, null, 0);
755     LocalVariable varPojo1 = new LocalVariable("instance", pojoClass, pojoDescriptor, 1);
756     LocalVariable varPojo2 = new LocalVariable("other", pojoClass, pojoDescriptor, 2);
757     LocalVariable varDifferencesList = new LocalVariable(
758       "differences", List.class, "Ljava/util/List<Lorg/pojomatic/diff/Difference;>;", 3);
759 
760     StackAdjustments stackAdjustments = new StackAdjustments();
761     Object[] localVarTypes = new Object[] {
762       pojomatorInternalClassName, OBJECT_INTERNAL_NAME, OBJECT_INTERNAL_NAME, internalName(List.class), null, null };
763 
764     MethodVisitor mv = cw.visitMethod(
765       ACC_PUBLIC, "doDiff", methodDesc(Differences.class, Object.class, Object.class), null, null);
766     mv.visitCode();
767     Label start = visitNewLabel(mv);
768     varPojo1.acceptLoad(mv);
769     visitLineNumber(mv, 50, null);
770     checkNotNull(mv, "instance is null");
771     varPojo2.acceptLoad(mv);
772     checkNotNull(mv, "other is null");
773 
774     // If instance and other are the same object, then return NoDifferences.getInstance();
775     Label notSameInstance = new Label();
776     mv.visitJumpInsn(IF_ACMPNE, notSameInstance);
777     mv.visitMethodInsn(INVOKESTATIC, internalName(NoDifferences.class), "getInstance", methodDesc(NoDifferences.class));
778     mv.visitInsn(ARETURN);
779 
780     // not the same instance, some work to do
781     mv.visitLabel(notSameInstance);
782     mv.visitFrame(F_FULL, 3, localVarTypes, 0, NO_STACK);
783     visitLineNumber(mv, 51, null);
784     checkCompatibleForEquality(mv, varThis, varPojo1, "instance");
785     visitLineNumber(mv, 52, null);
786     checkCompatibleForEquality(mv, varThis, varPojo2, "other");
787 
788     Label makeDiferences = notSameInstance;
789     mv.visitTypeInsn(NEW, "java/util/ArrayList");
790     mv.visitInsn(DUP);
791     mv.visitMethodInsn(INVOKESPECIAL, "java/util/ArrayList", "<init>", "()V");
792     varDifferencesList.acceptStore(mv);
793 
794     List<LocalVariable> propertyVariables = new ArrayList<>(); // these will occur in a block scope
795     // compare properties
796     for(PropertyElement propertyElement: classProperties.getHashCodeProperties()) {
797       int width = isWide(propertyElement) ? 2 : 1;
798       Class<?> propertyType = propertyElement.getPropertyType();
799       LocalVariable varProp1 = new LocalVariable(
800         "property_" + propertyElement.getName() + "_1", propertyType, null, 4);
801       //If the type is long or double, we need to store the next var at 6, not 5.
802       LocalVariable varProp2 = new LocalVariable(
803         "property_" + propertyElement.getName() + "_2", propertyType, null, 4 + width);
804       propertyVariables.add(varProp1);
805       propertyVariables.add(varProp2);
806 
807       Label blockStart = visitNewLabel(mv);
808 
809       visitLineNumber(mv, 53, propertyElement);
810       visitAccessor(mv, varPojo1, propertyElement);
811       varProp1.acceptStore(mv);
812       visitLineNumber(mv, 54, propertyElement);
813       visitAccessor(mv, varPojo2, propertyElement);
814       varProp2.acceptStore(mv);
815 
816       visitLineNumber(mv, 55, propertyElement);
817       visitAccessorAndConvert(mv, varPojo1, propertyElement);
818       visitLineNumber(mv, 56, propertyElement);
819       visitAccessorAndConvert(mv, varPojo2, propertyElement);
820 
821       Label propertiesNotEqual = new Label();
822       Label next = new Label();
823       visitLineNumber(mv, 57, propertyElement);
824       compareProperties(mv, propertiesNotEqual, propertyElement, stackAdjustments);
825       mv.visitJumpInsn(GOTO, next); // there were no differences.
826 
827       mv.visitLabel(propertiesNotEqual);
828 
829       localVarTypes[5] = localVarTypes[4] = propertyType.isPrimitive()
830         ? Primitives.getOpcode(propertyType)
831         : internalName(effectiveType(propertyType));
832       mv.visitFrame(F_FULL, 6, localVarTypes, 0, NO_STACK);
833 
834       // Create a ValueDifference instance, initialized with the property name and the two values, and add it to our list
835       varDifferencesList.acceptLoad(mv); // we'll need this to add to the list
836       mv.visitTypeInsn(NEW, "org/pojomatic/diff/ValueDifference");
837       mv.visitInsn(DUP);
838       mv.visitLdcInsn(propertyElement.getName());
839       varProp1.acceptLoad(mv);
840       visitLineNumber(mv, 58, propertyElement);
841       convertToObject(mv, propertyType);
842       varProp2.acceptLoad(mv);
843       visitLineNumber(mv, 59, propertyElement);
844       convertToObject(mv, propertyType);
845       visitLineNumber(mv, 60, propertyElement);
846       mv.visitMethodInsn(
847         INVOKESPECIAL,
848         internalName(ValueDifference.class),
849         "<init>",
850         methodDesc(void.class, String.class, Object.class, Object.class));
851 
852       // add the ValueDifference instance to our list
853       visitLineNumber(mv, 61, propertyElement);
854       mv.visitMethodInsn(INVOKEINTERFACE, internalName(List.class), "add", methodDesc(boolean.class, Object.class));
855       mv.visitInsn(POP); // ignore the return value of List#add
856       mv.visitLabel(next);
857       mv.visitFrame(F_FULL, 4, localVarTypes, 0, NO_STACK);
858 
859       varProp1.withScope(blockStart, next);
860       varProp2.withScope(blockStart, next);
861     }
862 
863     // if our list is empty, return the NoDifferences instance
864     varDifferencesList.acceptLoad(mv);
865     visitLineNumber(mv, 62, null);
866     mv.visitMethodInsn(INVOKEINTERFACE, internalName(List.class), "isEmpty", "()Z");
867     Label hasDifferences = new Label();
868     mv.visitJumpInsn(IFEQ, hasDifferences);
869     visitLineNumber(mv, 63, null);
870     mv.visitMethodInsn(INVOKESTATIC, internalName(NoDifferences.class), "getInstance", methodDesc(NoDifferences.class));
871     mv.visitInsn(ARETURN);
872 
873     // our list is not empty, so wrap it in a PropertyDiferences instance
874     mv.visitLabel(hasDifferences);
875     mv.visitFrame(F_FULL, 4, localVarTypes, 0, NO_STACK);
876 
877     visitLineNumber(mv, 64, null);
878     mv.visitTypeInsn(NEW, internalName(PropertyDifferences.class));
879     mv.visitInsn(DUP);
880     varDifferencesList.acceptLoad(mv);
881     visitLineNumber(mv, 65, null);
882     mv.visitMethodInsn(
883       INVOKESPECIAL, internalName(PropertyDifferences.class), "<init>", methodDesc(void.class, List.class));
884     mv.visitInsn(ARETURN);
885 
886     Label end = visitNewLabel(mv);
887 
888     varThis.withScope(start, end).acceptLocalVariable(mv);
889     varPojo1.withScope(start, end).acceptLocalVariable(mv);
890     varPojo2.withScope(start, end).acceptLocalVariable(mv);
891     varDifferencesList.withScope(makeDiferences, end).acceptLocalVariable(mv);
892     for (LocalVariable var: propertyVariables) {
893       var.acceptLocalVariable(mv);
894     }
895     mv.visitMaxs(6 + stackAdjustments.adjustments(2, 0), 6 + stackAdjustments.adjustments(2, 0));
896     mv.visitEnd();
897   }

918   /**
919    * Determine if the given propertyElement should be treated as possibly containing a multi-level array.
920    * This will be the case if it is:
921    * <ul>
922    *   <li>of type Object and is not annotated with @{@link SkipArrayCheck}</li>
923    *   <li>of type Object[]</li>
924    *   <li>of array type with a component type of array type</li>
925    * </ul>
926    * @param propertyElement
927    * @return {@code true} if the given propertyElement should be treated as possibly containing a multi-level array,
928    * or {@code false} otherwise.
929    */
930   private boolean isDeepArray(PropertyElement propertyElement) {
931     Class<?> propertyType = propertyElement.getPropertyType();
932     return
933       propertyType.equals(Object[].class)
934       || (propertyType.isArray() && propertyType.getComponentType().isArray())
935       || ((propertyType.equals(Object.class) || propertyType.equals(Object[].class))
936           && !propertyElement.getElement().isAnnotationPresent(SkipArrayCheck.class));
937   }

943    * with @{@link SkipArrayCheck}

947           && ! propertyElement.getElement().isAnnotationPresent(SkipArrayCheck.class);

998    * is not visible to us, we'll treat any type which is neither primitive nor an array to be of type Object.
