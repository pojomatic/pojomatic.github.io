6   /**
7    * Default formatter for classes that use {@link Pojomatic}.  This implementation first presents
8    * the class name, and then each property in turn, separated by commas, using braces to indicate
9    * nesting.
10   * <p>
11   * For example, if a class Person has two properties, firstName and LastName, and these properties
12   * are using {@link DefaultPropertyFormatter}, then the Person
13   * instance representing Joe Blow would be represented as
14   * <code>"Person{firstName: {Joe}, lastName: {Blow}}"</code>
15   */
16  public class DefaultPojoFormatter implements PojoFormatter {
17    private boolean firstPropertyPrinted = false;
18  
19    public String getPropertyPrefix(PropertyElement property) {
20      StringBuilder result = new StringBuilder();
21      if (firstPropertyPrinted) {
22        result.append(", ");
23      }
24      else {
25        firstPropertyPrinted = true;
26      }
27      return result.append(property.getName()).append(": {").toString();
28    }
29  
30    public String getPropertySuffix(PropertyElement property) {
31      return "}";
32    }
33  
34    public String getToStringPrefix(Class<?> pojoClass) {
35      return pojoClass.getSimpleName() + "{";
36    }
37  
38    public String getToStringSuffix(Class<?> pojoClass) {
39      return "}";
40    }
41  }
