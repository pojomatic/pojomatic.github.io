265   @Test public void testIsCompatibleForEquals() {
266     class Parent { @SuppressWarnings("unused") @Property int getX() { return 3; } }
267     class NonContributingChild extends Parent {}
268     @OverridesEquals class AnnotatedNonContributingChild extends Parent {}
269     class ContributingChild extends Parent{ @SuppressWarnings("unused") @Property int getY() { return 3; } }
270     class ChildOfContributingChild extends ContributingChild{}
271 
272     @SuppressWarnings("unchecked")
273     List<List<Class<?>>> partitions = Arrays.asList(
274       Arrays.<Class<?>>asList(Parent.class, NonContributingChild.class),
275       Arrays.<Class<?>>asList(ContributingChild.class, ChildOfContributingChild.class),
276       Arrays.<Class<?>>asList(AnnotatedNonContributingChild.class),
277       Arrays.<Class<?>>asList(Interface.class));
278     for (List<Class<?>> partition: partitions) {
279       for (Class<?> clazz: partition) {
280         for(List<Class<?>> otherPartition: partitions) {
281           for(Class<?> otherClazz: otherPartition) {
282             if (partition == otherPartition) {
283               assertTrue(ClassProperties.forClass(clazz).isCompatibleForEquals(otherClazz));
284             }
285             else {
286               assertFalse(ClassProperties.forClass(clazz).isCompatibleForEquals(otherClazz));
287             }
288           }
289         }
290       }
291     }
292   }
